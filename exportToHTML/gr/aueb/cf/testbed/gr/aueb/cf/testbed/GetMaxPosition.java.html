package gr.aueb.cf.testbed;

public class GetMaxPosition {

    public static void main(String[] args) {
        int[] grades = {4, 8, 10, 4, 6, 2, 1, 4};
        int maxPosition = 0;

        maxPosition = getMaxPosition(grades, 0, grades.length - 1);

        System.out.printf("Max Value; %d, Max position: %d", grades[maxPosition], maxPosition + 1);
    }


    // METHODOS
    public static int getMaxPosition (int[] arr, int low, int high) {
        int maxPosition = high;
        int maxValue;

        if ((low < 0) || (high > arr.length - 1)) {
            System.out.println("Error in array dimensions");
            return -1;
        }

        maxValue = arr[high];
        for (int i = high; i >= low; i--) {
            if (arr[i] > maxValue) {
                maxPosition = i;
                maxValue = arr[i];
            }
        }

        return maxPosition;
    }
}
